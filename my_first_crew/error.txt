        # 以MySQL为例：
        mysql -u${DB_ADMIN_USER} -p${DB_ADMIN_PASS} -h${DB_HOST} -e "SHOW FULL PROCESSLIST;" | grep -i orders | awk '{print "KILL", $1";"}' > /tmp/kill_slow_queries.sql
        
        # 2. 如果找到慢查询，终止它们。执行前请确认，避免误杀重要业务查询。
